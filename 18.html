import numpy as np
import pandas as pd
from sklearn.metrics import mean_squared_error, mean_absolute_error
import matplotlib.pyplot as plt
from statsmodels.graphics.tsaplots import plot_acf, plot_pacf

results = []
for order, model in fitted_models.items():
    predictions = model.predict(start=0, end=len(df) - 1)
    aic = model.aic
    bic = model.bic
    rmse = np.sqrt(mean_squared_error(df['logDPI'], predictions))
    mae = mean_absolute_error(df['logDPI'], predictions)
    results.append([order, aic, bic, rmse, mae])

results_df = pd.DataFrame(results, columns=['Order', 'AIC', 'BIC', 'RMSE', 'MAE'])
display(results_df)

# Residual analysis for each model

    residuals = model.resid
    plot_acf(residuals, ax=axes[i, 0], lags=20, title=f'ACF of Residuals (Order: {order})')
    plot_pacf(residuals, ax=axes[i, 1], lags=20, title=f'PACF of Residuals (Order: {order})')
    axes[i, 2].hist(residuals, bins=20, density=True, alpha=0.7, label='Residuals')

    axes[i, 2].legend()
